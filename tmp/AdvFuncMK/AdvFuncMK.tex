\section{Polynomial Functions}
\begin{theorem} [Factorial]
    $n! = n \times (n-1) \times (n-2) \times \dots \times 2 \times 1$
\end{theorem}
%
\begin{theorem}[Finite Differences]
    $\Delta y = a \times n!$
\end{theorem}
%
\begin{theorem}[Polynomial Transformation Formula]
    $P(x)=a[k(x-d)]^n + c$
\end{theorem}
